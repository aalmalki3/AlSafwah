// Code.gs const ss = SpreadsheetApp.getActiveSpreadsheet(); // FIXED: Gets the bound spreadsheet const SHEET_NAME = 'Workers'; function doGet(e) { return handleRequest(e); } function doPost(e) { return handleRequest(e); } function handleRequest(e) { const action = e.parameter.action; try { switch(action) { case 'getAll': return getAllWorkers(); case 'getById': return getWorkerById(e.parameter.id); case 'add': return addWorker(e); case 'update': return updateWorker(e); case 'delete': return deleteWorker(e.parameter.id); case 'checkIn': return checkInWorker(e.parameter.id); case 'checkOut': return checkOutWorker(e.parameter.id); case 'getStats': return getStatistics(); default: return createResponse('error', 'Invalid action'); } } catch (error) { return createResponse('error', error.message); } } function getAllWorkers() { const sheet = ss.getSheetByName(SHEET_NAME); const data = sheet.getDataRange().getValues(); const headers = data[0]; const workers = data.slice(1).map((row, index) => { let obj = { rowId: index + 2 }; headers.forEach((header, i) => { obj[header] = row[i]; // Keeps spaces in keys like "Arrival Date" }); return obj; }); return createResponse('success', 'Data retrieved successfully', workers); } function getWorkerById(id) { const sheet = ss.getSheetByName(SHEET_NAME); const data = sheet.getDataRange().getValues(); const worker = data.find(row => row[0] == id); if (!worker) return createResponse('error', 'Worker not found'); return createResponse('success', 'Worker found', worker); } function addWorker(e) { const sheet = ss.getSheetByName(SHEET_NAME); const data = JSON.parse(e.postData.contents); const newRow = [ data.id || generateId(), data.country, data.name, parseDate(data.arrivalDate), // FIXED: Convert to Date object parseDate(data.endDate), // FIXED: Convert to Date object calculateRemainingDays(data.endDate), data.section, data.status || 'Pending', data.reason, data.agency, data.comments, '', '', new Date().toISOString(), new Date().toISOString() ]; sheet.appendRow(newRow); return createResponse('success', 'Worker added successfully'); } function updateWorker(e) { const sheet = ss.getSheetByName(SHEET_NAME); const data = JSON.parse(e.postData.contents); const id = data.id; const rows = sheet.getDataRange().getValues(); let rowIndex = -1; for (let i = 1; i < rows.length; i++) { if (rows[i][0] == id) { rowIndex = i + 1; break; } } if (rowIndex === -1) return createResponse('error', 'Worker not found'); const r = rows[rowIndex-1]; // Existing row data const updatedRow = [ id, data.country || r[1], data.name || r[2], parseDate(data.arrivalDate || r[3]), // FIXED parseDate(data.endDate || r[4]), // FIXED calculateRemainingDays(data.endDate || r[4]), data.section || r[6], data.status || r[7], data.reason || r[8], data.agency || r[9], data.comments || r[10], r[11], r[12], r[13], new Date().toISOString() ]; sheet.getRange(rowIndex, 1, 1, updatedRow.length).setValues([updatedRow]); return createResponse('success', 'Worker updated successfully'); } function deleteWorker(id) { const sheet = ss.getSheetByName(SHEET_NAME); const rows = sheet.getDataRange().getValues(); for (let i = 1; i < rows.length; i++) { if (rows[i][0] == id) { sheet.deleteRow(i + 1); return createResponse('success', 'Worker deleted successfully'); } } return createResponse('error', 'Worker not found'); } function checkInWorker(id) { const sheet = ss.getSheetByName(SHEET_NAME); const rows = sheet.getDataRange().getValues(); for (let i = 1; i < rows.length; i++) { if (rows[i][0] == id) { sheet.getRange(i + 1, 12).setValue(new Date()); // FIXED: Use Date object sheet.getRange(i + 1, 15).setValue(new Date()); return createResponse('success', 'Worker checked in'); } } return createResponse('error', 'Worker not found'); } function checkOutWorker(id) { const sheet = ss.getSheetByName(SHEET_NAME); const rows = sheet.getDataRange().getValues(); for (let i = 1; i < rows.length; i++) { if (rows[i][0] == id) { sheet.getRange(i + 1, 13).setValue(new Date()); // FIXED sheet.getRange(i + 1, 15).setValue(new Date()); return createResponse('success', 'Worker checked out'); } } return createResponse('error', 'Worker not found'); } function getStatistics() { const sheet = ss.getSheetByName(SHEET_NAME); const data = sheet.getDataRange().getValues(); const stats = { total: 0, byCountry: {}, byStatus: {}, bySection: {}, byAgency: {}, checkInToday: 0, checkOutToday: 0 }; const today = new Date().toDateString(); data.slice(1).forEach(row => { stats.total++; const country = row[1]; stats.byCountry[country] = (stats.byCountry[country] || 0) + 1; const status = row[7]; stats.byStatus[status] = (stats.byStatus[status] || 0) + 1; const section = row[6]; stats.bySection[section] = (stats.bySection[section] || 0) + 1; const agency = row[9]; if (agency) stats.byAgency[agency] = (stats.byAgency[agency] || 0) + 1; if (row[11] && new Date(row[11]).toDateString() === today) stats.checkInToday++; if (row[12] && new Date(row[12]).toDateString() === today) stats.checkOutToday++; }); return createResponse('success', 'Statistics retrieved', stats); } function createResponse(status, message, data = null) { return ContentService.createTextOutput(JSON.stringify({ status, message, data, timestamp: new Date().toISOString() })).setMimeType(ContentService.MimeType.JSON); } function calculateRemainingDays(endDate) { if (!endDate) return 0; const end = new Date(endDate); const now = new Date(); const diffDays = Math.ceil((end - now) / (1000 * 60 * 60 * 24)); return diffDays > 0 ? diffDays : 0; } function generateId() { return 'W' + Date.now().toString(36).toUpperCase(); } // Helper to ensure dates are saved as Date objects, not strings function parseDate(dateStr) { if (!dateStr) return ''; if (dateStr instanceof Date) return dateStr; return new Date(dateStr); }
